    import { describe, it, expect, vi, beforeEach } from 'vitest';
    import { renderHook, waitFor } from '@testing-library/react';
    import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
    import { useCreatePoll } from './use-create-poll';
    import * as contractService from '../midnight/contract-service';
    import { createMockProviders } from '../test/test-utils';

    // Mock the contract service
    vi.mock('../midnight/contract-service', () => ({
      deployPollContract: vi.fn(),
    }));

    const createWrapper = () => {
      const queryClient = new QueryClient({
        defaultOptions: { queries: { retry: false } },
      });
      return ({ children }: { children: React.ReactNode }) => (
        <QueryClientProvider client={queryClient}>{children}</QueryClientProvider>
      );
    };

    describe('useCreatePoll', () => {
      const mockProviders = createMockProviders();

      beforeEach(() => {
        vi.clearAllMocks();
      });

      it('should create poll successfully', async () => {
        const mockPollId = new Uint8Array(32).fill(1);
        vi.mocked(contractService.deployPollContract).mockResolvedValue({
          address: 'midnight1poll...',
          pollId: mockPollId,
        });

        const { result } = renderHook(
          () => useCreatePoll(mockProviders),
          { wrapper: createWrapper() }
        );

        result.current.mutate({
          title: 'Test Poll',
          description: 'Test Description',
          options: ['Option 1', 'Option 2'],
          pollType: 0,
          durationDays: 7,
        });

        await waitFor(() => expect(result.current.isSuccess).toBe(true));
        expect(result.current.data?.pollId).toEqual(mockPollId);
      });

      it('should handle poll creation error', async () => {
        vi.mocked(contractService.deployPollContract).mockRejectedValue(
          new Error('Wallet not connected')
        );

        const { result } = renderHook(
          () => useCreatePoll(mockProviders),
          { wrapper: createWrapper() }
        );

        result.current.mutate({
          title: 'Test Poll',
          description: 'Test',
          options: ['A', 'B'],
          pollType: 0,
          durationDays: 7,
        });

        await waitFor(() => expect(result.current.isError).toBe(true));
        expect(result.current.error?.message).toBe('Wallet not connected');
      });

      it('should show loading state while creating', async () => {
        vi.mocked(contractService.deployPollContract).mockImplementation(
          () => new Promise((resolve) => setTimeout(resolve, 100))
        );

        const { result } = renderHook(
          () => useCreatePoll(mockProviders),
          { wrapper: createWrapper() }
        );

        result.current.mutate({
          title: 'Test Poll',
          description: 'Test',
          options: ['A'],
          pollType: 0,
          durationDays: 1,
        });

        expect(result.current.isPending).toBe(true);
      });
    });

    import { describe, it, expect, vi } from 'vitest';
    import { renderHook, waitFor } from '@testing-library/react';
    import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
    import { useMetadata } from './use-metadata';

      const queryClient = new QueryClient({

      });

      );
    };

    describe('useMetadata', () => {
      it('should fetch metadata for poll', async () => {
        const mockMetadata = {
          title: 'Test Poll',
          description: 'Test Description',
          options: ['Option 1', 'Option 2'],
          createdAt: Date.now(),
        };

        vi.mocked(metadataStore.getMetadata).mockResolvedValue(mockMetadata);

        const pollId = new Uint8Array(32).fill(1);
        const { result } = renderHook(
          () => useMetadata(pollId),
          { wrapper: createWrapper() }
        );

        await waitFor(() => expect(result.current.isSuccess).toBe(true));
        expect(result.current.data).toEqual(mockMetadata);
      });

      it('should handle missing metadata', async () => {
        vi.mocked(metadataStore.getMetadata).mockResolvedValue(null);

        const pollId = new Uint8Array(32).fill(99);
        const { result } = renderHook(
          () => useMetadata(pollId),
          { wrapper: createWrapper() }
        );

    import { renderHook, waitFor } from '@testing-library/react';

    import { usePoll } from './use-poll';
    import * as indexerClient from '../midnight/indexer-client';

      const queryClient = new QueryClient({
        defaultOptions: { queries: { retry: false } },
      });

        <QueryClientProvider client={queryClient}>{children}</QueryClientProvider>
      );
    };

    describe('usePoll', () => {
      it('should fetch poll by ID', async () => {
        const mockPoll = {
          id: new Uint8Array(32).fill(1),
          titleHash: new Uint8Array(32).fill(2),
          creator: 'midnight1creator...',
          optionsCount: 3,
          pollType: 0,
          expirationBlock: 1000n,
        };

        vi.mocked(indexerClient.getPoll).mockResolvedValue(mockPoll);

        const pollId = 'poll123';
        const { result } = renderHook(
          () => usePoll(pollId),
          { wrapper: createWrapper() }
        );

        await waitFor(() => expect(result.current.isSuccess).toBe(true));
        expect(result.current.data).toEqual(mockPoll);
      });

      it('should handle poll not found', async () => {
        vi.mocked(indexerClient.getPoll).mockResolvedValue(null);

        const pollId = 'nonexistent';
        const { result } = renderHook(
          () => usePoll(pollId),
          { wrapper: createWrapper() }
        );

        await waitFor(() => expect(result.current.isSuccess).toBe(true));
        expect(result.current.data).toBeNull();
      });

      it('should show loading state', async () => {
        vi.mocked(indexerClient.getPoll).mockImplementation(
          () => new Promise((resolve) => setTimeout(() => resolve({} as any), 50))

          () => usePoll(pollId),
          { wrapper: createWrapper() }
        );
