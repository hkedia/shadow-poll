    ```typescript
    import { TestEnvironment, TestSession } from '@midnight-ntwrk/testkit-js';
    import { Contract } from '@midnight-ntwrk/compact-runtime';
    
    export async function setupTestEnvironment(): Promise<TestEnvironment> {
      // Initialize testkit-js simulator environment
      const env = new TestEnvironment();
      await env.initialize();
      return env;
    }
    
    export function createTestUser(env: TestEnvironment) {
      return env.createUser();
    }
    
    export async function deployPollContract(
      session: TestSession,
      // Contract deployment parameters
    ) {
      // Load compiled contract
      const contract = await Contract.fromCompiled(
        './contracts/managed/poll_contract.json'
      );
      
      // Deploy to simulator
      const deployment = await session.deploy(contract);
      return deployment;
    }
    
    export function generatePollId(
      title: string,
      creator: string,
      timestamp: number
    ): Uint8Array {
      // Generate deterministic poll ID for testing
      const data = `${title}-${creator}-${timestamp}`;
      return crypto.subtle.digest('SHA-256', new TextEncoder().encode(data));
    }
    
    export function generateInviteCodeHash(
      pollId: Uint8Array,
      code: string
    ): Uint8Array {
      // Hash invite code with poll ID
      const data = new Uint8Array([...pollId, ...new TextEncoder().encode(code)]);
      return crypto.subtle.digest('SHA-256', data);
    }
    
    export function bytesToHex(bytes: Uint8Array): string {
      return Array.from(bytes)
        .map(b => b.toString(16).padStart(2, '0'))
        .join('');
    }
    
    export function hexToBytes(hex: string): Uint8Array {
      const bytes = new Uint8Array(hex.length / 2);
      for (let i = 0; i < hex.length; i += 2) {
        bytes[i / 2] = parseInt(hex.substr(i, 2), 16);
      }
      return bytes;
    }
    ```
