    ```typescript
    import { describe, it, expect } from 'vitest';
    import {
      cn,
      formatDate,
      truncateAddress,
      formatBlockNumber,
      classNames,
    } from './utils';

    describe('utils', () => {
      describe('cn (classNames utility)', () => {
        it('should merge class names', () => {
          const result = cn('class1', 'class2');
          expect(result).toBe('class1 class2');
        });

        it('should handle conditional classes', () => {
          const result = cn('base', true && 'active', false && 'inactive');
          expect(result).toBe('base active');
        });

        it('should handle object syntax', () => {
          const result = cn('base', { active: true, disabled: false });
          expect(result).toBe('base active');
        });

        it('should filter out falsy values', () => {
          const result = cn('base', null, undefined, '', 'valid');
          expect(result).toBe('base valid');
        });

        it('should merge tailwind classes correctly', () => {
          const result = cn('px-2 py-1', 'px-4');
          // tailwind-merge should deduplicate
          expect(result).toContain('px-4');
          expect(result).not.toContain('px-2');
        });
      });

      describe('formatDate', () => {
        it('should format date to readable string', () => {
          const date = new Date('2026-04-10T12:00:00Z');
          const result = formatDate(date);
          expect(result).toContain('2026');
          expect(result).toContain('Apr');
        });

        it('should handle timestamp numbers', () => {
          const timestamp = 1712745600000; // 2024-04-10
          const result = formatDate(timestamp);
          expect(typeof result).toBe('string');
          expect(result.length).toBeGreaterThan(0);
        });

        it('should handle string dates', () => {
          const result = formatDate('2026-04-10');
          expect(typeof result).toBe('string');
        });
      });

      describe('truncateAddress', () => {
        it('should truncate long addresses', () => {
          const address = 'midnight1abcdefghijklmnopqrstuvwxyz123';
          const result = truncateAddress(address);
          expect(result.length).toBeLessThan(address.length);
          expect(result).toContain('...');
        });

        it('should preserve short addresses', () => {
          const address = 'short';
          const result = truncateAddress(address);
          expect(result).toBe(address);
        });

        it('should handle custom length', () => {
          const address = 'midnight1verylongaddressthatneedstruncating';
          const result = truncateAddress(address, 8);
          expect(result.length).toBeLessThanOrEqual(19); // 8 + ... + 8
        });

        it('should handle empty string', () => {
          const result = truncateAddress('');
          expect(result).toBe('');
        });
      });

      describe('formatBlockNumber', () => {
        it('should format block number with commas', () => {
          const result = formatBlockNumber(1234567n);
          expect(result).toBe('1,234,567');
        });

        it('should handle zero', () => {
          const result = formatBlockNumber(0n);
          expect(result).toBe('0');
        });

        it('should handle large numbers', () => {
          const result = formatBlockNumber(1000000000000n);
          expect(result).toContain(',');
        });
      });

      describe('classNames (legacy)', () => {
        it('should still work for backward compatibility', () => {
          const result = classNames('a', 'b', { c: true });
          expect(result).toContain('a');
          expect(result).toContain('b');
          expect(result).toContain('c');
        });
      });
    });
    ```
