    ```typescript
    import { describe, it, expect, vi } from 'vitest';
    import { Hono } from 'hono';
    import { healthHandler } from './health-handler';
    import * as dbClient from '../db/client';

    vi.mock('../db/client', () => ({
      sql: vi.fn(),
    }));

    describe('health-handler', () => {
      const app = new Hono();
      app.get('/api/health', healthHandler);

      it('should return 200 when database is healthy', async () => {
        vi.mocked(dbClient.sql).mockResolvedValue([{ now: new Date() }]);

        const res = await app.request('/api/health');
        expect(res.status).toBe(200);

        const body = await res.json();
        expect(body).toEqual({ status: 'healthy' });
      });

      it('should return 503 when database is unreachable', async () => {
        vi.mocked(dbClient.sql).mockRejectedValue(new Error('Connection failed'));

        const res = await app.request('/api/health');
        expect(res.status).toBe(503);

        const body = await res.json();
        expect(body).toEqual({ status: 'degraded' });
      });

      it('should include CORS headers', async () => {
        vi.mocked(dbClient.sql).mockResolvedValue([{ now: new Date() }]);

        const res = await app.request('/api/health');
        expect(res.headers.get('Access-Control-Allow-Origin')).toBe('*');
      });
    });
    ```

    ```typescript
    import { describe, it, expect, vi, beforeEach } from 'vitest';
    import { Hono } from 'hono';
    import { getMetadataHandler, postMetadataHandler } from './metadata-handler';
    import * as dbClient from '../db/client';

    vi.mock('../db/client', () => ({
      sql: vi.fn(),
    }));

    describe('metadata-handler', () => {
      const app = new Hono();
      app.get('/api/polls/metadata', getMetadataHandler);
      app.post('/api/polls/metadata', postMetadataHandler);

      beforeEach(() => {
        vi.clearAllMocks();
      });

      describe('GET /api/polls/metadata', () => {
        it('should return metadata for specific poll', async () => {
          const mockMetadata = {
            poll_id: 'poll123',
            title: 'Test Poll',
            description: 'Test Description',
            options: JSON.stringify(['Option 1', 'Option 2']),
            created_at: new Date().toISOString(),
          };

          vi.mocked(dbClient.sql).mockResolvedValue([mockMetadata]);

          const res = await app.request('/api/polls/metadata?pollId=poll123');
          expect(res.status).toBe(200);

          const body = await res.json();
          expect(body).toHaveProperty('poll_id', 'poll123');
          expect(body).toHaveProperty('title', 'Test Poll');
        });

        it('should return 404 for non-existent poll', async () => {
          vi.mocked(dbClient.sql).mockResolvedValue([]);

          const res = await app.request('/api/polls/metadata?pollId=nonexistent');
          expect(res.status).toBe(404);
        });

        it('should return all polls when no pollId provided', async () => {
          const mockPolls = [
            { poll_id: 'poll1', title: 'Poll 1' },
            { poll_id: 'poll2', title: 'Poll 2' },
          ];

          vi.mocked(dbClient.sql).mockResolvedValue(mockPolls);

          const res = await app.request('/api/polls/metadata');
          expect(res.status).toBe(200);

          const body = await res.json();
          expect(Array.isArray(body)).toBe(true);
          expect(body).toHaveLength(2);
        });
      });

      describe('POST /api/polls/metadata', () => {
        it('should create new metadata', async () => {
          vi.mocked(dbClient.sql).mockResolvedValue([{ poll_id: 'newpoll' }]);

          const res = await app.request('/api/polls/metadata', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              pollId: 'newpoll',
              title: 'New Poll',
              description: 'Description',
              options: ['A', 'B', 'C'],
            }),
          });

          expect(res.status).toBe(201);

          const body = await res.json();
          expect(body).toHaveProperty('poll_id', 'newpoll');
        });

        it('should update existing metadata (idempotent)', async () => {
          vi.mocked(dbClient.sql)
            .mockResolvedValueOnce([{ poll_id: 'existing' }]) // Check exists
            .mockResolvedValueOnce([{ poll_id: 'existing' }]); // Update

          const res = await app.request('/api/polls/metadata', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              pollId: 'existing',
              title: 'Updated Title',
              description: 'Updated Description',
              options: ['X', 'Y'],
            }),
          });

          expect(res.status).toBe(200);
        });

        it('should validate required fields', async () => {
          const res = await app.request('/api/polls/metadata', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ title: 'Missing pollId' }),
          });

          expect(res.status).toBe(400);
        });
      });
    });
    ```
