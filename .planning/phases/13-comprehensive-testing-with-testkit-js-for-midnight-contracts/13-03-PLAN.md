            address: 'midnight1contract...',

            status: 'connected',
            address: 'midnight1creator...',

          const result = await deployPollContract(providers, {
            titleHash: new Uint8Array(32).fill(1),
            descriptionHash: new Uint8Array(32).fill(2),
            optionsCount: 3,
            pollType: 0,
            expirationBlock: 1000n,

          expect(result).toBeDefined();
        });

            status: 'disconnected',

              titleHash: new Uint8Array(32),
              descriptionHash: new Uint8Array(32),
              optionsCount: 3,
              pollType: 0,

    import { describe, it, expect } from 'vitest';
    import {
      generateInviteCode,
      inviteCodeToBytes32,
      validateInviteCodeFormat,
    } from './invite-codes';

    describe('invite-codes', () => {
      describe('generateInviteCode', () => {
        it('should generate code of specified length', () => {
          const code = generateInviteCode(8);
          expect(code.length).toBe(8);
        });

        it('should generate unique codes', () => {
          const code1 = generateInviteCode(10);
          const code2 = generateInviteCode(10);
          expect(code1).not.toBe(code2);
        });

        it('should only contain alphanumeric characters', () => {
          const code = generateInviteCode(12);
          expect(code).toMatch(/^[A-Z0-9]+$/);
        });
      });

      describe('inviteCodeToBytes32', () => {
        it('should convert code to 32-byte array', () => {
          const bytes = inviteCodeToBytes32('TESTCODE');
          expect(bytes.length).toBe(32);
        });

        it('should produce consistent results for same code', () => {
          const bytes1 = inviteCodeToBytes32('SAMECODE');
          const bytes2 = inviteCodeToBytes32('SAMECODE');
          expect(bytes1).toEqual(bytes2);
        });

        it('should handle case insensitively', () => {
          const bytes1 = inviteCodeToBytes32('testcode');
          const bytes2 = inviteCodeToBytes32('TESTCODE');
          expect(bytes1).toEqual(bytes2);
        });
      });

      describe('validateInviteCodeFormat', () => {
        it('should accept valid alphanumeric codes', () => {
          expect(validateInviteCodeFormat('ABC123')).toBe(true);
          expect(validateInviteCodeFormat('VALID8')).toBe(true);
        });

        it('should reject codes with special characters', () => {
          expect(validateInviteCodeFormat('ABC-123')).toBe(false);
          expect(validateInviteCodeFormat('CODE!')).toBe(false);
        });

        it('should reject empty codes', () => {
          expect(validateInviteCodeFormat('')).toBe(false);
        });

        it('should reject codes that are too short', () => {
          expect(validateInviteCodeFormat('AB')).toBe(false);
        });
      });
    });

          const key1 = computeTallyKey(pollId, 0);

          expect(key1).toEqual(key2);

        it('should produce different keys for different options', () => {
          const pollId = new Uint8Array(32).fill(1);
          const key0 = computeTallyKey(pollId, 0);
          const key1 = computeTallyKey(pollId, 1);
          expect(key0).not.toEqual(key1);
        });

        it('should produce different keys for different polls', () => {
          const pollId1 = new Uint8Array(32).fill(1);
          const pollId2 = new Uint8Array(32).fill(2);
          const key1 = computeTallyKey(pollId1, 0);
          const key2 = computeTallyKey(pollId2, 0);
          expect(key1).not.toEqual(key2);
        });
      });

      describe('persistentHash', () => {
        it('should produce 32-byte hash', async () => {
          const data = new TextEncoder().encode('test data');
          const hash = await persistentHash(data);
          expect(hash.length).toBe(32);
        });

        it('should produce consistent hash for same data', async () => {
          const data = new TextEncoder().encode('consistent');
          const hash1 = await persistentHash(data);
          const hash2 = await persistentHash(data);
          expect(hash1).toEqual(hash2);
        });

        it('should produce different hashes for different data', async () => {
          const hash1 = await persistentHash(new TextEncoder().encode('data1'));
          const hash2 = await persistentHash(new TextEncoder().encode('data2'));

        });
      });

          expect(date).toBeInstanceOf(Date);
          expect(date.getTime()).toBeGreaterThan(Date.now());
        });

        it('should handle same block number', () => {
          const block = 1000n;
          const date = blockNumberToDate(block, block);
          expect(date.getTime()).toBeLessThanOrEqual(Date.now() + 1000);
        });

          const targetBlock = 6n; // 1 minute
          const date = blockNumberToDate(currentBlock, targetBlock);
          const expectedMs = 6 * 10 * 1000; // 6 blocks * 10 seconds * 1000ms
          expect(date.getTime() - Date.now()).toBeCloseTo(expectedMs, -2);
        });
      });
